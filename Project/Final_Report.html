class VAE(keras.Model):

    def __init__(self):
        super(VAE, self).__init__()

        self.layer1 = layers.Dense(128)
        self.layer2 = layers.Dense(z_dim)
        self.layer3 = layers.Dense(z_dim)

        self.layer4 = layers.Dense(128)
        self.layer5 = layers.Dense(784)

    def encoder(self, x):

        h = tf.nn.relu(self.layer1(x))

        mu = self.layer2(h)

        log_var = self.layer3(h)

        return mu, log_var

    def decoder(self, z):

        output = tf.nn.sigmoid(self.layer4(z))
        output = self.layer5(tf.nn.sigmoid(output))

        return output

    def reparameterize(self, mu, log_var):

        eps = tf.random.normal(log_var.shape)

        std = tf.math.exp(log_var)**0.5

        z = mu + std * eps
        return z

    def call(self, inputs, training=None):

        mu, log_var = self.encoder(inputs)

        z = self.reparameterize(mu, log_var)

        x_hat = self.decoder(z)

        return x_hat,mu,z,log_var

def CB_log_norm(lam, approx = 0.01):
  lam = tf.clip_by_value(lam, 1e-06, 1.0-(1e-06))
  # if |lam-0.5| < approx, use taylor approximation to avoid numeric instability
  log_norm = tf.math.log(1e-6 + 2.0*tf.math.atanh(1.0-2.0*lam)/(1.0-2.0*lam))
  taylor_approx = tf.math.log(2.0) + 4/3*tf.pow(lam-0.5, 2) + 104/45*tf.pow(lam-0.5,4) + (64/7-64/15+64/81)*tf.pow(lam-0.5,6)
  return tf.where(tf.math.abs(lam-0.5)>=approx, log_norm, taylor_approx)

def CB_mean(lam, approx = 0.01):
  # if |lam-0.5| < approx, use taylor approximation to avoid numeric instability
  mean =  lam/(2.0*lam-1.0) + 1.0/(2.0*tf.math.atanh(1.0-2.0*lam))
  taylor_approx =  0.5 + 1/3*(lam-0.5) + 16/45*tf.pow(lam-0.5,3) + 704/945*tf.pow(lam-0.5,5)
  return tf.where(tf.math.abs(lam-0.5)>=approx, mean, taylor_approx)

class CB_VAE(keras.Model):
  def __init__(self, latent_dim=20, n_hid=500, dropout_rate=0.1):
    super(CB_VAE, self).__init__()
    self.latent_dim = latent_dim
    self.n_hid = n_hid
    self.dropout_rate = dropout_rate

    # encoder NN
    self.inference = keras.Sequential(
        [ keras.layers.Conv2D(filters=32, kernel_size=(3,3), padding='same', 
                              input_shape=(28,28,1), activation='relu'),
          keras.layers.MaxPool2D(),
          keras.layers.Flatten(),
          keras.layers.Dense(n_hid, activation='relu'),
          keras.layers.Dropout(dropout_rate),
          keras.layers.Dense(n_hid, activation='relu'),
          keras.layers.Dropout(dropout_rate),
          keras.layers.Dense(2*latent_dim)
        ]
    )

    # decoder NN
    self.generative = keras.Sequential(
        [ 
          keras.layers.InputLayer(input_shape=(latent_dim,)),
          keras.layers.Dense(n_hid, activation='relu'),
          keras.layers.Dropout(dropout_rate),
          keras.layers.Dense(n_hid, activation='relu'),
          keras.layers.Dropout(dropout_rate),
          keras.layers.Dense(784, activation='sigmoid')
        ]
    )

  @tf.function
  def sample(self, z, source = 'lambda'):
    lams = self.decode(z)
    if source == 'lambda':
      samples = lams
    
    elif source == 'mean':
      samples = CB_mean(lams)

    elif source == 'CB':
      """
      Random sample from CB using inverse transform
      """
      u = tf.random.uniform(shape=[784], minval=0, maxval=1, dtype='float32')
      samples = tf.where(tf.math.abs(lams-0.5)<approx, u, 
                        (tf.math.log(u*(2.0*lams-1)+1.0-lams)-tf.math.log(1.0-lams)) / ((tf.math.log(lams)-tf.math.log(1.0-lams))))
      
    if len(z.shape) == 2:
        samples = tf.reshape(samples, (z.shape[0],28,28))
    else:
        samples = tf.reshape(samples, (28,28))
    return samples

  def encode(self, x):
    gaussian_params = self.inference(x)
    mean = gaussian_params[:,:self.latent_dim]
    #mean = tf.math.reduce_mean(mean, 0)
    sd = tf.math.softplus(gaussian_params[:,self.latent_dim:])
    #sd = tf.math.reduce_mean(sd, 0)
    return mean, sd
  
  def reparameterize(self, mean, sd):
    eps = tf.random.normal(mean.shape)
    return eps * sd + mean
  
  def decode(self, z):
    lam = self.generative(z)
    return lam

        log_p_all = tf.math.reduce_sum(x*tf.math.log(lams) + (1.0-x)*tf.math.log(1.0-lams) + log_norm_const, 1)
        log_p = tf.math.reduce_mean(log_p_all)
        
        KL = 0.5 * tf.math.reduce_sum(tf.pow(mean,2) + tf.pow(sd,2) - tf.math.log(tf.pow(sd,2))-1.0, 1)
        KL = tf.math.reduce_mean(KL)
        
        elbo = log_p - KL
        obj = -elbo # maximize ELBO -> minimize obj

    gradients = tape.gradient(obj, model_CB.trainable_variables)
    for i in range(len(gradients)):
      gradients[i] = tf.clip_by_value(gradients[i], -2, 2)
    optimizer.apply_gradients(zip(gradients, model_CB.trainable_variables))
    train_elbo(elbo)
    with train_summary_writer.as_default():
      tf.summary.scalar('ELBO', train_elbo.result(), step=epoch)

  end_time = time.time()
  print('Epoch: ', epoch)
  print('######## Training time: %f' % (end_time-start_time), 's ########')

  if epoch % 50 == 1 or epoch == 200:
    image = tf.reshape(model_CB.sample(rdm_vector_for_generation),(-1,28,28))
    gen_image(image, epoch, 'CB')

def cnn_model(input_shape = (28,28,1)):
  model = tf.keras.Sequential()
  model.add(tf.keras.layers.Conv2D(32, kernel_size= (3, 3),padding="same",data_format="channels_last",input_shape = input_shape))
  model.add(tf.keras.layers.MaxPooling2D((2, 2)))
  model.add(tf.keras.layers.Flatten())
  model.add(tf.keras.layers.Dense(128, activation='relu', kernel_initializer='he_uniform'))
  model.add(tf.keras.layers.Dense(10, activation='softmax'))
  model.compile(optimizer="adam", loss='sparse_categorical_crossentropy', metrics=['accuracy'])
  return model
